```typescript
async function fetchGitLabSprints(
  repoId: string,
): Promise<{ sprints: DashboardSprint[]; issues: DashboardIssue[] }> {
  try {
    const milestones = await gitlabClient.listMilestones(repoId, "active");
    const sprints: DashboardSprint[] = milestones.map((m: any) => ({
      id: `gl-milestone-${m.id}`,
      name: m.title || `Milestone ${m.id}`,
      state: m.state === "active" ? "active" : "closed",
      startDate: m.start_date || m.created_at || "",
      endDate: m.due_date || "",
      totalPoints: 0,
      completedPoints: 0,
      platform: "gitlab",
      repo: repoId,
    }));

    // Fetch all issues and map sprint field from milestone property
    let issues: DashboardIssue[] = [];
    try {
      const rawIssues = await gitlabClient.listIssues(repoId, "all");
      issues = rawIssues.map((i: any) => {
        const milestone = i.milestone;
        const sprint: string | undefined = milestone
          ? `gl-milestone-${milestone.id}`
          : undefined;
        return {
          id: String(i.iid || i.id),
          externalId: `!${i.iid || i.id}`,
          title: i.title || "",
          url: i.web_url || "",
          status: normalizeStatus(i.state, "gitlab", i.labels || []),
          priority: normalizePriority(null, i.labels || []),
          assignee: i.assignee?.username || i.assignee?.name || null,
          labels: i.labels || [],
          platform: "gitlab",
          repo: repoId,
          createdAt: i.created_at || "",
          updatedAt: i.updated_at || "",
          dependencies: parseDependencies(i.description || ""),
          sprint,
        };
      });
    } catch {
      /* fall through with empty issues */
    }

    // Compute point totals from issue counts (same pattern as GitHub/Jira)
    for (const sprint of sprints) {
      const sprintIssues = issues.filter((i) => i.sprint === sprint.id);
      sprint.totalPoints = sprintIssues.length;
      sprint.completedPoints = sprintIssues.filter(
        (i) => i.status === "done",
      ).length;
    }

    return { sprints, issues };
  } catch (err: any) {
    return { sprints: [], issues: [] };
  }
}
```

**Key differences from `fetchGitHubSprints`:**
- GitLab returns `milestone.id` (numeric DB ID), not `milestone.number`. We use `gl-milestone-{id}`.
- GitLab milestones have `start_date` for sprint start, while GitHub uses `created_at`.
- The `state` field: GitLab uses `"active"`/`"closed"`, which maps to `"active"`/`"closed"` directly.

function resetSprintModal() {
  sprintModalName.value = "";
  sprintModalDesc.value = "";
  sprintModalStart.value = "";
  sprintModalEnd.value = "";
  sprintModalMode = "create";
  editingSprintId = null;
  sprintModalTitle.textContent = "Create Sprint / Milestone";
  sprintModalConfirm.textContent = "Create";
}

function showSprintModal(mode, existingSprint) {
  resetSprintModal();
  if (mode === "edit" && existingSprint) {
    sprintModalMode = "edit";
    editingSprintId = existingSprint.id;
    sprintModalTitle.textContent = "Edit Sprint / Milestone";
    sprintModalConfirm.textContent = "Save Changes";
    sprintModalName.value = existingSprint.name || "";
    sprintModalStart.value = existingSprint.startDate
      ? new Date(existingSprint.startDate).toISOString().split("T")[0]
      : "";
    sprintModalEnd.value = existingSprint.endDate
      ? new Date(existingSprint.endDate).toISOString().split("T")[0]
      : "";
  }
  sprintModal.style.display = "";
}

function hideSprintModal() {
  sprintModal.style.display = "none";
  resetSprintModal();
}

function hideSprintDeleteModal() {
  sprintDeleteModal.style.display = "none";
}

async function submitSprintForm() {
  var repoVal = repoSelect.value;
  var colon = repoVal.indexOf(":");
  var platform = repoVal.slice(0, colon);
  var repoKey = repoVal.slice(colon + 1);

  var name = sprintModalName.value.trim();
  if (!name) {
    showError("Sprint/milestone name is required.");
    return;
  }

  var body = {
    platform: platform,
    repo: repoKey,
    name: name,
    description: sprintModalDesc.value.trim() || undefined,
    startDate: sprintModalStart.value
      ? new Date(sprintModalStart.value).toISOString()
      : undefined,
    endDate: sprintModalEnd.value
      ? new Date(sprintModalEnd.value).toISOString()
      : undefined,
  };

  try {
    var url, method;
    if (sprintModalMode === "edit" && editingSprintId) {
      url = "/api/repo-dashboard/sprints/" + encodeURIComponent(editingSprintId);
      method = "PUT";
    } else {
      url = "/api/repo-dashboard/sprints";
      method = "POST";
    }

    var resp = await fetch(url, {
      method: method,
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(body),
    });

    if (!resp.ok) {
      var errData = await resp.json().catch(function () { return { error: "HTTP " + resp.status }; });
      throw new Error(errData.error || "Request failed");
    }

    hideSprintModal();
    // Reload sprints
    loadSprints(platform, repoKey);
  } catch (err) {
    showError("Failed to save sprint/milestone: " + err.message);
  }
}

async function confirmDeleteSprint() {
  var repoVal = repoSelect.value;
  var colon = repoVal.indexOf(":");
  var platform = repoVal.slice(0, colon);
  var repoKey = repoVal.slice(colon + 1);

  var sprintId = sprintSelect.value;
  if (!sprintId) return;

  try {
    var resp = await fetch(
      "/api/repo-dashboard/sprints/" + encodeURIComponent(sprintId),
      {
        method: "DELETE",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ platform: platform, repo: repoKey }),
      }
    );

    if (!resp.ok) {
      var errData = await resp.json().catch(function () { return { error: "HTTP " + resp.status }; });
      throw new Error(errData.error || "Delete failed");
    }

    hideSprintDeleteModal();
    currentSprintId = null;
    destroyChart(BURNDOWN_CANVAS_ID);
    sprintStatsBar.style.display = "none";
    renderTable();
    loadSprints(platform, repoKey);
  } catch (err) {
    showError("Failed to delete sprint/milestone: " + err.message);
  }
}

## Error Handling Patterns

All new code must follow the project's established error handling conventions:

1. **Client methods**: Use try/catch that re-throws with `axios.isAxiosError()` checks for status-code-specific messages (404, 400, 401, 403), falling through to a generic `Error` with the original message. This matches the existing pattern in all three clients.

2. **Dashboard routes**: Catch errors with `catch (err: any)` and return 500 with `{ error: err.message || "Failed to..." }`. This matches the existing `/sprints` and `/burndown` route error pattern.

3. **Tool dispatcher**: Return `{ success: false, error: err.message }` with a try/catch. This matches the existing `handleGithubCreateIssue` etc. pattern.

4. **Frontend**: Use `try/catch` with `resp.ok` checks and fallback JSON parsing, calling `showError()` on failure. This matches the existing `loadSprints` and `loadBurndown` patterns.

5. **Cache invalidation**: After write operations (create/update/delete), call `invalidateIssueCache(platform, repo)` to ensure the next GET /issues fetch returns fresh data. This follows the existing `invalidateIssueCache` pattern already exported from repo-dashboard.ts.
